type: object
required:
  - name
  - gender
  - idNumber
  - birthDate
  - employmentDate
  - checkDate
  - companyName
  - companyAddress
  - pastExperience
  - currentJob
  - dailyWorkingHours
  - checkPeriod
  - chronicDiseases

  - betelNutChewingHabit
  - drinkingHabit
  - symptoms

  name:
    type: string
    title: 姓名
  gender:
    type: string
    title: 性別

    title: 身分證字號 (護照號碼)

    title: 檢查日期

    title: 事業單位名稱 (廠別)

    type: string
    title: 地址

    type: object

    type: object

        type: string

        type: string
        title: 起始日期
        pattern: "\\d{3}年\\d{1,2}月"
      endDate:
        type: string
        title: 截至
        pattern: "\\d{3}年\\d{1,2}月"
      duration:
        type: string
        title: 共計
        pattern: "\\d{1,2}年\\d{1,2}月"

    title: 從事黃磷作業平均每日工時

      - const: 變更作業

      - const: 定期檢查
        title: 定期檢查
      - const: 健康追蹤檢查
        title: 健康追蹤檢查

          type: string

            - 慢性氣管炎、肺氣腫

            - 無

        type: array
        title: 肝臟疾病

          type: string
          enum:
            - B型肝炎

            - 酒精性肝炎
            - 藥物性肝炎

        type: object
        title: (幾乎)每天吸

            type: integer
            title: 已吸菸

        type: object
        title: 已經戒菸

            type: integer

          averagePerDay:
            type: integer

          yearsChewed:
            type: integer
            title: 已嚼

        type: object
        title: 已經戒食
        properties:
          yearsQuit:
            type: integer
            title: 戒了
          monthsQuit:
            type: integer
            title: 個月

    type: object
    title: 生活習慣 - 喝酒
    properties:
      neverDrank:
        type: boolean
        title: 從未喝酒
      occasionallyDrinks:
        type: boolean
        title: 偶爾喝(不是天天)
      drinksDaily:
        type: object
        title: （幾乎）每天喝
        properties:
          averagePerWeek:
            type: integer
            title: 平均每週喝

            type: string
            title: 最常喝
          amountPerOccasion:
            type: integer
            title: 每次
      quitDrinking:
        type: object
        title: 已經戒酒
        properties:
          yearsQuit:
            type: integer
            title: 戒了
          monthsQuit:
            type: integer
            title: 個月

    type: object

        type: array
        title: 心臟血管
        items:
          type: string
          enum:
            - 心悸
            - 頭暈
            - 頭痛
      respiratorySystem:
        type: array
        title: 呼吸系統
        items:
          type: string
          enum:
            - 咳嗽
            - 咳痰
            - 呼吸困難
            - 胸痛

        type: array

          type: string

        items:
          type: string
          enum:

        title: 其他
        items:
          type: string

            - 下顎痛
